#include <stdio.h>
#include <stdlib.h>
typedef struct Node { int data; struct Node *next; } Node;
Node *removeDup(Node *head) {
    Node *cur = head;
    while (cur) {
        Node *prev = cur, *p = cur->next;
        while (p) {
            if (p->data == cur->data) {
                prev->next = p->next;
                free(p);
                p = prev->next;
            } else { prev = p; p = p->next; }
        }
        cur = cur->next;
    }
    return head;
}
